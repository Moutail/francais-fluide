💼 Comment Utiliser Ces Logos

Intégration dans Next.js

Créez un fichier Logo.tsx dans src/components/ui/ :

// src/components/ui/Logo.tsx import React from 'react'; interface LogoProps { variant?: 'full' | 'icon' | 'text'; className?: string; width?: number; height?: number; } export const Logo: React.FC<LogoProps> = ({ variant = 'full', className = '', width = 200, height = 60 }) => { if (variant === 'icon') { return ( <svg width="40" height="40" viewBox="0 0 40 40" className={className}> <rect x="0" y="0" width="40" height="40" rx="8" fill="#2563eb" /> <text x="20" y="26" fontFamily="Inter" fontSize="18" fontWeight="700" fill="white" textAnchor="middle"> FF </text> </svg> ); } return ( <svg width={width} height={height} viewBox="0 0 200 60" className={className}> <defs> <linearGradient id="logoGrad" x1="0%" y1="0%" x2="100%" y2="0%"> <stop offset="0%" style={{ stopColor: '#2563eb', stopOpacity: 1 }} /> <stop offset="100%" style={{ stopColor: '#3b82f6', stopOpacity: 1 }} /> </linearGradient> </defs> <g transform="translate(10, 10)"> <rect x="0" y="0" width="40" height="40" rx="8" fill="url(#logoGrad)" /> <text x="20" y="30" fontFamily="Inter" fontSize="20" fontWeight="700" fill="white" textAnchor="middle"> FF </text> </g> <text x="60" y="35" fontFamily="Inter" fontSize="20" fontWeight="500" fill="#1e293b"> FrançaisFluide </text> </svg> ); };

Utilisation dans le Header

// Dans votre Header.tsx import { Logo } from '@/components/ui/Logo'; <header className="bg-white border-b"> <div className="flex items-center"> <Logo variant="full" className="h-10" /> {/* ou pour mobile */} <Logo variant="icon" className="md:hidden" /> </div> </header>

Favicon et Meta Tags

Pour le favicon, utilisez la version icône (FF monogramme) et générez les différentes tailles :

  • favicon.ico (32x32)
  • icon-192.png (192x192) pour PWA
  • icon-512.png (512x512) pour PWA
  • apple-touch-icon.png (180x180) pour iOS
